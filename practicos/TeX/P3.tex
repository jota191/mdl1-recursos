\begin{ejercicio}
  Determinar cuáles de las siguientes propiedades verifican las funciones
  que se definen a continuación:
\begin{itemize}
\item
inyectividad
\item
sobreyectividad
\item
biyectividad
\end{itemize}

\begin{enumerate}
\item
  $f: \mathbb{Z} \to \mathbb{Z}$ definida como $f(z)=z-1$.
\item
  $f: \mathbb{N} \times \mathbb{N} \to \mathbb{N}$ definida como
  $f(x,y)=x$.
\item
  $f: \mathbb{N} \times \mathbb{N} \to \mathbb{N}$ definida como
  $f(x,y)=x+y$.
\item
  $f: \mathbb{N} \times \mathbb{N} \to \mathbb{N}$ definida como
  $f(x,y)=xy$.
\item
  $f: \mathbb{N} \times \mathbb{N} \to \mathbb{N}$ definida como
  $f(x,y)=x+y+1$.
\item
  $f: \mathbb{N} \to \mathbb{N}$ definida como $f(x)=2x $.


\item
  $f: \mathbb{N} \times \mathbb{N} \to \mathbb{N}$ definida como
  $f(x,y)=2^x * 3^y$.


\end{enumerate}

\end{ejercicio}


\begin{ejercicio}
  Se consideran conjuntos $A$ y $B$,
  con $|A|=n$ y $|B|=m$, $m,n\in \mathbb{N}$.
\begin{enumerate}
\item
  ¿Cuántas funciones de $A$ en $B$ se pueden definir?
\item
  ¿Qué relación deben cumplir $m$ y $n$ para poder definir una
  función inyectiva
  de $A$ en $B$?\\
  ¿Cuántas funciones inyectivas de $A$ en $B$ se pueden definir?
\item
  ¿Qué relación deben cumplir $m$ y $n$ para poder definir una función
  sobreyectiva de $A$ en $B$? ¿Qué relación deben cumplir $m$ y $n$ para
  poder definir una función biyectiva de $A$ en $B$?\\
  ¿Cuántas funciones biyectivas de $A$ en $B$ se pueden definir?
\end{enumerate}
\end{ejercicio}

\begin{ejercicio}
  Probar que la función de $f:\mathbb{N}\to P$,
  donde $P$ es el conjunto de los números pares
  (es decir, $\{2n: n \in \mathbb{N}\}$) definida como $f(n)=2n$ es biyectiva.
  ¿Qué podría decirse de la cantidad de elementos de estos conjuntos?
\end{ejercicio}

\begin{ejercicio}
  Para los siguientes pares de funciones
  $f,g:\mathbb{Z} \rightarrow \mathbb{Z}$
  calcular $f \circ g$ y $g \circ f$. 
  \begin{enumerate}
  \item $f(x) = x$, $g(x) = x$
  \item $f(x) = x^2$, $g(x) = 4x$
  \item $f(x) = 2x + 5$, $g(x) = 3x - 1$
  \item $f(x) = 2x - 1$, $g(x) = x^{3} - x^{2} - 4$
  \end{enumerate}
\end{ejercicio}


\begin{ejercicio}
  Consideramos las operaciones $\oplus$ y $\otimes$
  en $ \{ 0,1\}$, definidas como:

  $$a \oplus b = \left \{ \begin{matrix} 0 & \mbox{si } a = b = 0
    \\ 1 & \mbox{en otro caso} \end{matrix}\right. , \hspace{1cm}
  a \otimes b = \left \{ \begin{matrix} 1 & \mbox{si } a = b = 1
    \\  0 & \mbox{en otro caso} \end{matrix}\right.$$

  Determinar si tienen neutro y encontrarlo.
\end{ejercicio}
